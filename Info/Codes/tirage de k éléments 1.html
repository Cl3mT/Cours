int* select(int* tab, int n, int k) {
    int* res = malloc(k * sizeof(int));
    for (int i = 0; i < k; i += 1) {
        res[i] = tab[k];
    }
    for (int i = k; i < n; i += 1) {
        int x = rand() % (i + 1);
        if (x < k) {
            res[x] = tab[i];
        }
    }
    return res;
}
